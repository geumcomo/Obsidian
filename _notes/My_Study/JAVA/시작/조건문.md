# 문제 1
```java
package cond.ex;  
  
public class ScoreEx {  
    public static void main(String[] args) {  
        int score = 60;  
        char grade = 'N';  
        if(score >= 90)  
            grade = 'A';  
        else if (score >= 80 && score < 90)  
            grade = 'B';  
        else if (score >= 70 && score < 80)  
            grade = 'C';  
        else if (score >= 60 && score < 70)  
            grade = 'D';  
        else  
            grade = 'F';  
  
        System.out.println("score: " + score );  
        System.out.println("출력: 학점은 " + grade + "입니다." );  
        }  
  
}
```

# 문제 5
```java
package cond.ex;  
  
public class GradeSwitchEx {  
    public static void main(String[] args) {  
        String grade = "B";  
  
        switch (grade){  
            case "A":  
                grade = "'A':탁월한 성과입니다!";  
                break;  
            case "B":  
                grade = "'B':좋은 성과입니다!";  
                break;  
            case "C":  
                grade = "'C':준수한 성과입니다!";  
                break;  
            case "D":  
                grade = "'D':향상이 필요합니다.";  
                break;  
            case "F":  
                grade = "'F':불합격입니다!";  
                break;  
            default:  
                grade = "잘못된 학점입니다.";  
        }  
        System.out.println(grade);  
    }  
}
```
